# Blink_SOS
## USE the arduino

```c
void setup() {
  pinMode(LED_BUILTIN, OUTPUT);
}


void loop() {
  digitalWrite(LED_BUILTIN, HIGH);   
  delay(200);                      
  digitalWrite(LED_BUILTIN, LOW);   
  delay(200); 
  digitalWrite(LED_BUILTIN, HIGH);  
  delay(200);  
  digitalWrite(LED_BUILTIN, LOW);   
  delay(200); 
  digitalWrite(LED_BUILTIN, HIGH);   
  delay(200);
  digitalWrite(LED_BUILTIN, LOW);
  delay(200);
  digitalWrite(LED_BUILTIN,HIGH);    
  delay(700);

  digitalWrite(LED_BUILTIN,HIGH);    
  delay(700);

  delay(700);  
  digitalWrite(LED_BUILTIN,HIGH);    
  delay(700);
  digitalWrite(LED_BUILTIN,LOW);
  delay(200);    
   digitalWrite(LED_BUILTIN, HIGH);

  digitalWrite(LED_BUILTIN, LOW);   
  delay(200);
```
